$def with (record, filename, begin, length) $var title: $_("MARC details of") $filename:$begin:$length $ end = begin + length - 1

$_("MARC details of") $filename:$begin:$length

$_("Record Locator:") $filename

$if not record:

$_("Invalid MARC record.")

$else:

LEADER: $record.leader
$:record.html()

$_("In order to retrieve the exact bytes of this MARC record, perform a HTTP GET on the url") 

http://www.archive.org/download/$filename
$_("and include the HTTP header") 
Range: bytes=$begin-$end

$:_('You will need to do this with a special purpose web client such as curl, not a browser. The curl command you\'d use is:') 

curl -L -r $begin-$end 'http://www.archive.org/download/$filename'
